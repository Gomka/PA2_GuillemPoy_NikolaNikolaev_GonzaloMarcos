public class SushiMonitor_01 {

	int emptySeats = 5
	boolean groupEating = false
	
	Condition canEnter

	public void enter(int i) {
		
		while(emptySeats <= 0 || groupEating) {

			if (emptySeats <= 0)
				groupEating = true

			waitC(canEnter)
		}
		
		add customer to table
		decrease emptySeats

		signalC(canEnter)
	}

	public void exit(int i) {

		increase emptySeats

		if (emptySeats >= 5)
			groupEating = false
		
		signalC(canEnter)
	}
}
